jQuery AJAX Label Example

Simple Example:

No Name

HTML:

<span class="ajaxLabel" id="example1" data-url="response.json">
	<span class="ajaxLabelView">
		No Name
	</span>
	<span class="ajaxLabelEdit">
		<input type="text" id="name" name="name"/>                        
	</span>
</span>

JS:

<script>
$(function() {
 $("#example1").ajaxLabel({
   success: function (response, model) {					
	model.viewPane.children().remove();
	model.viewPane.text(model.editPane.find("#name").val());					
   }
 });
});
</script>

Validation Example:

Your name is Unknown

HTML:

<span class="ajaxLabel" id="example2" data-url="response.json">
	<span class="ajaxLabelView">
		Your name is Unknown
	</span>
	<span class="ajaxLabelEdit">
		<input type="text" id="name" name="name"/>                        
	</span>
</span>

JS:

<script>
$(function() {
 $("#example2").ajaxLabel({
  validate: function (model) {
	var name = model.editPane.find("#name").val();
	if(name == "" || name == null)
		return { valid: false, message: "Name is required!" };
	return { valid: true };
   },
   success: function (response, model) {					
	model.viewPane.children().remove();
	model.viewPane.text(model.editPane.find("#name").val());					
   }
 });
});
</script>
